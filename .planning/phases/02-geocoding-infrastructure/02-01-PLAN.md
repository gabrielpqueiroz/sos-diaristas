**1. Create `scripts/migrate-geocoding-columns.js`:**

A standalone Node.js script that adds three columns to crm_orders. Use `process.loadEnvFile('.env.local')` (Node 20.12+, project is Node 24) to load DATABASE_URL. Connect via `pg` directly (not via src/lib/db.js since this runs outside Next.js).

```javascript
// scripts/migrate-geocoding-columns.js
process.loadEnvFile('.env.local')
import pg from 'pg'
const { Pool } = pg

const pool = new Pool({ connectionString: process.env.DATABASE_URL })

const SQL = `
ALTER TABLE crm_orders
  ADD COLUMN IF NOT EXISTS lat NUMERIC,
  ADD COLUMN IF NOT EXISTS lng NUMERIC,
  ADD COLUMN IF NOT EXISTS geocoded_at TIMESTAMPTZ;
`

async function migrate() {
  console.log('Running migration: add lat, lng, geocoded_at to crm_orders...')
  await pool.query(SQL)
  console.log('Migration complete.')
  await pool.end()
}

migrate().catch(err => { console.error('Migration failed:', err.message); process.exit(1) })
```

Use `IF NOT EXISTS` to make it idempotent. Column types: `NUMERIC` for lat/lng (avoids floating-point surprises), `TIMESTAMPTZ` for geocoded_at.

Create `scripts/geocode-backfill.js` — a standalone Node.js script run with `node scripts/geocode-backfill.js`.

**Structure:**

```javascript
// scripts/geocode-backfill.js
process.loadEnvFile('.env.local')
import pg from 'pg'
import { geocodeAddress } from '../src/lib/geocode.js'

const { Pool } = pg
const pool = new Pool({ connectionString: process.env.DATABASE_URL })
```

**Rate limiter** — hand-rolled, 1 req/s minimum gap:
```javascript
function createThrottle(minGapMs = 1000) {
  let lastAt = 0
  return async function throttle() {
    const now = Date.now()
    const wait = minGapMs - (now - lastAt)
    if (wait > 0) await new Promise(r => setTimeout(r, wait))
    lastAt = Date.now()
  }
}
```

**Main logic:**
1. Query: `SELECT id, address FROM crm_orders WHERE lat IS NULL AND address IS NOT NULL ORDER BY created_at DESC`
2. Log: `"Found {N} orders to geocode"`
3. For each order: `await throttle()`, then `await geocodeAddress(order.address)`
4. If result is not null: `UPDATE crm_orders SET lat=$1, lng=$2, geocoded_at=NOW() WHERE id=$3` with params `[result.lat, result.lng, order.id]`
5. If result is null: increment failCount, log `"[SKIP] Order {id}: no result for '{address}'"`
6. Log progress every order: `"[{i}/{total}] {address} -> {lat},{lng}"` or `"[{i}/{total}] {address} -> NOT FOUND"`
7. Final summary: `"Done. Geocoded: {success}/{total} ({percent}%). Failed: {failCount} addresses."`
8. `await pool.end()` then `process.exit(0)`

**--dry-run flag:** If `process.argv.includes('--dry-run')`, skip the UPDATE query and log `"[DRY RUN]"` prefix. This allows testing without writing to DB.

**Error handling:** Wrap the main loop in try/catch. If any individual geocode or UPDATE fails, log the error and continue to the next order (do not abort the entire backfill).
